4 C> THIS SUBROUTINE STORES INFORMATION ABOUT A "CHILD"
5 C> MNEMONIC WITHIN THE INTERNAL BUFR TABLE D ENTRY (IN MODULE
6 C> TABABD) FOR A TABLE D SEQUENCE ("PARENT") MNEMONIC WHEN THE
7 C> "CHILD" MNEMONIC IS CONTAINED WITHIN THE SEQUENCE REPRESENTED BY
8 C> THE "PARENT" MNEMONIC (AS DETERMINED WITHIN BUFR ARCHIVE LIBRARY
9 C> SUBROUTINE SEQSDX).

54  SUBROUTINE pktdd(ID,LUN,IDN,IRET)
55 
56 
57 
58  USE modv_maxcd
59  USE moda_tababd
60 
61  include 'bufrlib.prm'
62 
63  COMMON /dxtab / maxdx,idxv,nxstr(10),ldxa(10),ldxb(10),ldxd(10),
64  . ld30(10),dxstr(10)
65  COMMON /quiet / iprt
66 
67  CHARACTER*128 ERRSTR
68  CHARACTER*56 DXSTR
69 
70 C-----------------------------------------------------------------------
71 C-----------------------------------------------------------------------
72 
73  ldd = ldxd(idxv+1)+1
74 
75 C LDD points to the byte within TABD(ID,LUN) which contains (in
76 C packed integer format) a count of the number of child mnemonics
77 C stored thus far for this parent mnemonic.
78 
79 C ZERO THE COUNTER IF IDN IS ZERO
80 C -------------------------------
81 
82  IF(idn.EQ.0) THEN
83  CALL ipkm(tabd(id,lun)(ldd:ldd),1,0)
84  iret = 0
85  GOTO 100
86  ENDIF
87 
88 C UPDATE THE STORED DESCRIPTOR COUNT FOR THIS TABLE D ENTRY
89 C ---------------------------------------------------------
90 
91  nd = iupm(tabd(id,lun)(ldd:ldd),8)
92 
93 C ND is the (unpacked) count of the number of child mnemonics
94 C stored thus far for this parent mnemonic.
95 
96  IF(nd.LT.0 .OR. nd.EQ.maxcd) THEN
97  IF(iprt.GE.0) THEN
98  CALL errwrt('+++++++++++++++++++++WARNING+++++++++++++++++++++++')
99  IF(nd.LT.0) THEN
100  WRITE ( unit=errstr, fmt='(A,I4,A)' )
101  . 'BUFRLIB: PKTDD - BAD COUNTER VALUE (=', nd,
102  . ') - RETURN WITH IRET = -1'
103  ELSE
104  WRITE ( unit=errstr, fmt='(A,I4,A,A)' )
105  . 'BUFRLIB: PKTDD - MAXIMUM NUMBER OF CHILD MNEMONICS (=',
106  . maxcd, ') ALREADY STORED FOR THIS PARENT - RETURN WITH ',
107  . 'IRET = -1'
108  ENDIF
109  CALL errwrt(errstr)
110  CALL errwrt('+++++++++++++++++++++WARNING+++++++++++++++++++++++')
111  CALL errwrt(' ')
112  ENDIF
113  iret = -1
114  GOTO 100
115  ELSE
116  nd = nd+1
117  CALL ipkm(tabd(id,lun)(ldd:ldd),1,nd)
118  iret = nd
119  ENDIF
120 
121 C PACK AND STORE THE DESCRIPTOR
122 C -----------------------------
123 
124  idm = ldd+1 + (nd-1)*2
125 
126 C IDM points to the starting byte within TABD(ID,LUN) at which
127 C the IDN value for this child mnemonic will be stored (as a
128 C packed integer of width = 2 bytes).
129 
130  CALL ipkm(tabd(id,lun)(idm:idm),2,idn)
131 
132 C EXIT
133 C ----
134 
135 100 RETURN
136  END
