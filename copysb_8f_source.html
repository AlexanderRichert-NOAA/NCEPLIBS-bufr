4 C> THIS SUBROUTINE COPIES A PACKED DATA SUBSET, INTACT, FROM
5 C> LOGICAL UNIT LUNIN, OPENED FOR INPUT VIA A PREVIOUS CALL TO BUFR
6 C> ARCHIVE LIBRARY SUBROUTINE OPENBF, TO LOGICAL UNIT LUNOT, OPENED
7 C> FOR OUTPUT VIA A PREVIOUS CALL TO OPENBF. THE BUFR MESSAGE MUST
8 C> HAVE BEEN PREVIOUSLY READ FROM UNIT LUNIT USING BUFR ARCHIVE
9 C> LIBRARY SUBROUTINE READMG OR READERME AND MAY BE EITHER COMPRESSED
10 C> OR UNCOMPRESSED. ALSO, BUFR ARCHIVE LIBRARY SUBROUTINE OPENMG OR
11 C> OPENMB MUST HAVE BEEN PREVIOUSLY CALLED TO OPEN AND INITIALIZE A
12 C> BUFR MESSAGE WITHIN MEMORY FOR UNIT LUNOT. EACH CALL TO COPYSB
13 C> ADVANCES THE POINTER TO THE BEGINNING OF THE NEXT SUBSET IN BOTH
14 C> THE INPUT AND OUTPUT FILES, UNLESS INPUT PARAMETER LUNOT IS .LE.
15 C> ZERO, IN WHICH CASE THE OUTPUT POINTER IS NOT ADVANCED. THE
16 C> COMPRESSION STATUS OF THE OUTPUT SUBSET/BUFR MESSAGE WILL ALWAYS
17 C> MATCH THAT OF THE INPUT SUBSET/BUFR MESSAGE {I.E., IF INPUT MESSAGE
18 C> IS UNCOMPRESSED(COMPRESSED) OUTPUT MESSAGE WILL BE UNCOMPRESSED
19 C> (COMPRESSED)}.
20 C>

100  IF(inode(lin).NE.inode(lot)) THEN

103  ENDIF
104  ENDIF
105 
106 C SEE IF THERE IS ANOTHER SUBSET IN THE MESSAGE
107 C ---------------------------------------------
108 
109  IF(nsub(lin).EQ.msub(lin)) THEN
110  iret = -1
111  GOTO 100
112  ENDIF
113 
114 C CHECK COMPRESSION STATUS OF INPUT MESSAGE, OUTPUT MESSAGE WILL MATCH
115 C --------------------------------------------------------------------
116 
117  CALL mesgbc(-lunin,mest,icmp)
118 
119  IF(icmp.EQ.1) THEN
120 
121 C -------------------------------------------------------
122 C THIS BRANCH IS FOR COMPRESSED INPUT/OUTPUT MESSAGES
123 C -------------------------------------------------------
124 C READ IN AND UNCOMPRESS SUBSET, THEN COPY IT TO COMPRESSED OUTPUT MSG
125 C --------------------------------------------------------------------
126 
127  CALL readsb(lunin,iret)
128  IF(lunot.GT.0) THEN
129  CALL ufbcpy(lunin,lunot)
130  CALL cmpmsg('Y')
131  CALL writsb(lunot)
132  CALL cmpmsg('N')
133  ENDIF
134  GOTO 100
135  ELSE IF(icmp.EQ.0) THEN
136 
137 C -------------------------------------------------------
138 C THIS BRANCH IS FOR UNCOMPRESSED INPUT/OUTPUT MESSAGES
139 C -------------------------------------------------------
140 C COPY THE SUBSET TO THE OUTPUT MESSAGE AND/OR RESET THE POINTERS
141 C ---------------------------------------------------------------
142 
143  ibit = (mbyt(lin))*8
144  CALL upb(nbyt,16,mbay(1,lin),ibit)
145  IF (nbyt.GT.65530) THEN
146 
147 C This is an oversized subset, so we can't rely on the value
148 C of NBYT as being the true size (in bytes) of the subset.
149 
150  IF ( (nsub(lin).EQ.0) .AND. (msub(lin).EQ.1) ) THEN
151 
152 C But it's also the first and only subset in the message,
153 C so we can determine its true size in a different way.
154 
155  CALL getlens(mbay(1,lin),4,len0,len1,len2,len3,len4,l5)
156  nbyt = len4 - 4
157  ELSE
158 
159 C We have no way to easily determine the true size of this
160 C oversized subset.
161 
162  iret = -1
163  GOTO 100
164  ENDIF
165  ENDIF
166  IF(lunot.GT.0) CALL cpyupd(lunot,lin,lot,nbyt)
167  mbyt(lin) = mbyt(lin) + nbyt
168  nsub(lin) = nsub(lin) + 1
169  ELSE
170  GOTO 907
171  ENDIF
172 
173 C EXITS
174 C -----
175 
176 100 RETURN
