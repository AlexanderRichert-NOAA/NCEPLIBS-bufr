66 C-----------------------------------------------------------------------
67 
68 C CONVERT USER NUMBERS INTO SCALED INTEGERS
69 C -----------------------------------------
70 
71  DO n=1,nval(lun)
72  node = inv(n,lun)
73  IF(itp(node).EQ.1) THEN
74  ival(n) = val(n,lun)
75  ELSEIF(typ(node).EQ.'NUM') THEN
76  IF(ibfms(val(n,lun)).EQ.0) THEN
77  ival(n) = ipks(val(n,lun),node)
78  ELSE
79  ival(n) = -1
80  ENDIF
81  ENDIF
82  ENDDO
83 
84 C PACK THE USER ARRAY INTO THE SUBSET BUFFER
85 C ------------------------------------------
86 
87  ibit = 16
88 
89  DO n=1,nval(lun)
90  node = inv(n,lun)
91  IF(itp(node).LT.3) THEN
92 
93 C The value to be packed is numeric.
94 
95  CALL pkb(ival(n),ibt(node),ibay,ibit)
96  ELSE
97 
98 C The value to be packed is a character string.
99 
100  ncr=ibt(node)/8
101  IF ( ncr.GT.8 .AND. luncpy(lun).NE.0 ) THEN
102 
103 C The string is longer than 8 characters and there was a
104 C preceeding call to UFBCPY involving this output unit, so
105 C read the long string with READLC and write it into the
106 C output buffer using PKC.
107 
108  CALL readlc(luncpy(lun),lstr,tag(node))
109  CALL pkc(lstr,ncr,ibay,ibit)
110  ELSE
111  rval = val(n,lun)
112  IF(ibfms(rval).NE.0) THEN
113 
114 C The value is "missing", so set all bits to 1 before
115 C packing the field as a character string.
116 
117  numchr = min(ncr,len(lstr))
118  DO jj = 1, numchr
119  CALL ipkm(lstr(jj:jj),1,255)
120  ENDDO
121  CALL pkc(lstr,numchr,ibay,ibit)
122  ELSE
123 
124 C The value is not "missing", so pack the equivalenced
125 C character string. Note that a maximum of 8 characters
126 C will be packed here, so a separate subsequent call to
127 C BUFR archive library subroutine WRITLC will be needed to
128 C fully encode any string longer than 8 characters.
129 
130  CALL pkc(cval,ncr,ibay,ibit)
131  ENDIF
132  ENDIF
133 
134  ENDIF
135  ENDDO
136 
137 C RESET UFBCPY FILE POINTER
138 C -------------------------
139 
140  luncpy(lun)=0
141 
142  RETURN
143  END
